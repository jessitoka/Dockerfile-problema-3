FROM ubuntu:20.04  #Esta linea de comando hace que desde el repositorio de Docker Hub se "descargue o utilice(?" la imagen en este caso, de ubuntu verificada por expertos, y os números indican la versión del ubuntu que se va a utilizar.

RUN apt-get update && apt-get install -y apache2 mysql-server  #Lo que ocurra en esta linea de comandos va siendo guardado en una nueva capa perteneciente a la imagen base de ubuntu que descargué en la anterior linea, y lo que va guardando en este caso es las actualizaciones de paquetes y si ese proceso es correcto, entonces tambien se comienzan a instalar automaticamente(leí que -y responde como un si, asi que entiendo que es automatico porque ya no pregunta como tal) el servidor web de Apache y MySQL para la bases de datos.

EXPOSE 80 3306  # Pues el dockerfile sabe que servicios estamos uasando pero no como acceder a ellos, seria como un indicador al contenedor desde donde va a estar recibiendo datos de esos servicios, en este caso 80  es el puerto por predeterminado que se utiliza para APACHE y el 3306 es de predeeterminado de MySQL.

CMD service apache2 start && service mysql start && tail -f /dev/null  # Entiendo que el CMD da como la orden ejecutar o iniciar los comandos cuando la imagen sea usada, entonces le da inicio a los servicios de APACHE y si esto anterior ocurre sin ningún inconveniente, entonces tambien comienza con el inicio de MySQL y todo esto ocurre indefinidamente debido a la parte finl del comando, entiendo que es como si se siguiera el finalde un archivo pero como ese archivo esta vacio, no tiene un final, asi que por eso se mantiene en ejecución.
